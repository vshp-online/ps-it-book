% Примеры остаются частью основного набора: без рамок и плашек, но с
% достаточным отступом, чтобы их начало не отделялось от текста страницы.

  \pretocmd{\example}{%
    \par\Needspace{6\baselineskip}\addvspace{10pt}%
  }{}{}%
  \apptocmd{\endexample}{%
    \par\addvspace{10pt}%
  }{}{}%
}

% Единый двухъячеечный блок для материалов к практическим примерам:
% QR-код слева, путь от корня репозитория справа.
\newcommand{\repolinkblock}[3]{%
  \par\medskip
  \begin{center}
    \small
    \setlength{\tabcolsep}{5pt}%
    \renewcommand{\arraystretch}{1}%
    \begin{tabular}{|c|l|}
      \hline
      \parbox[c][26mm][c]{25mm}{\centering\includegraphics[width=21mm]{#1}} &
      \parbox[c][26mm][c]{108mm}{\raggedright\href{#2}{\texttt{\detokenize{#3}}}} \\
      \hline
    \end{tabular}
  \end{center}
  \medskip
}
